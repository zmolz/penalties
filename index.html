date_pat = re.compile(r'(\d\d?)(\d\d)')

def parse_date(d):
    match = re.match(date_pat, str(d))
    month = '0'+match.group(1) if len(match.group(1)) == 1 else match.group(1)
    day = match.group(2)
    return f'{month}-{day}'

# create a closure with the regex and the dictionary and this function
def matchup_parser(matchup):
    match = re.match(matchup_pat, matchup)
    away = to_abv[match.group(1)]
    home = to_abv[match.group(2)]
    return home, away

# closure
def penalty_parser(penalty):
    pen_on = re.search(penalty_on_pat, penalty).group(1)
    pen = re.search(penalty_pat, penalty).group(1)
    pen = (pen.split(','))[-1]
    return pen_on, pen

# closure
def time_parser(time):
    time_match = re.match(time_pat, time)
    quarter = time_match.group(1)
    time = time_match.group(2)
    return quarter, time


for ref, games in ref_to_games.items():

As we look at the graph, we see something somewhat unexpected - as your team gets more penalties called on them, your team is expected to win more. However, this is where critical thinking comes into play in Data Science. The gamedata table includes information on games not just in the regular season, but in the playoffs as well, so those teams in the playoffs are getting more penalties called against them than other teams, and getting more wins as well. To adjust for this, we have to use a more reliable variable - win percentage would be a better gauge.
